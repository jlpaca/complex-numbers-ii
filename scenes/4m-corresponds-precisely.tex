  \newcommand\tf{8}

  \zgrid[unit=3, out=\tg+\dt]{

    \pgfmathsetmacro\arcorig\pgfmathresult

    \draw[bg-b, dashed] (\zunit, 0) arc (0:\arcorig:\zunit);    
    \pgfkeys{/pgf/fpu=true}
    
    \kfclip{0}{\ta+\dt}{
      \fill ({\zunit*cos(\za r)}, {\zunit*sin(\za r)}) circle (0.12);
    }

    \zin[in=\ta, out=\td+\dt, arg origin=pi/3]{1}{\za}
    \zout[in=\td+\dt, out=\tg+\dt, arg visible=false]{1}{pi}
    \zargout[in=\td+\dt, arg origin=pi/3, out=\td+2*\dt]{pi}

  }
